GetSelectedOutputArray Method

Loads a phreeqc database into the phreeqc object. Returns the number of errors encountered while loading the database.

expression.GetSelectedOutputArray()

expression    Required. An expression that returns a phreeqc object.

Example

This example runs the "ex2" input file and writes the selected output array to stdout.


  JScript

var Phreeqc = new ActiveXObject("PhreeqcCOM.Phreeqc");
Phreeqc.LoadDatabase("phreeqc.dat");
Phreeqc.RunFile("ex2");

// Note: JScript doesn't support multi-dimensional arrays
var arr = Phreeqc.GetSelectedOutputArray().toArray();

for (var i = 0; i < Phreeqc.RowCount; ++i)
{
    for (var j = 0; j < Phreeqc.ColumnCount; ++j)
    {
        WScript.Stdout.Write(arr[i + j*Phreeqc.RowCount]);
        WScript.Stdout.Write(" ");
    }
    WScript.Stdout.WriteLine();
}


  VBScript

Dim Phreeqc, arr, i, j
Set Phreeqc = CreateObject("PhreeqcCOM.Phreeqc")
Phreeqc.LoadDatabase("phreeqc.dat")
Phreeqc.RunFile("ex2")

arr = Phreeqc.GetSelectedOutputArray()
For i = 0 To Phreeqc.RowCount - 1
    For j = 0 To Phreeqc.ColumnCount - 1
        If (IsObject(WScript.Stdout)) Then
            WScript.Stdout.Write(arr(i, j) & " ")
        End If
    Next
    If (IsObject(WScript.Stdout)) Then
        WScript.Stdout.WriteLine
    End If
Next
