OutputStringOn Property

True to store data normally sent to the output file to a string buffer for retrieval. The initial setting is False. Read/write Boolean.

Remarks

The OutputStringOn property is used only by the Run methods: RunAccumulated, RunFile and RunString.

Example

This example equilibrates pure water with calcite and displays the results.


  Powershell

[System.IO.Directory]::SetCurrentDirectory($PSScriptRoot)
$iphreeqc = New-Object -ComObject IPhreeqcCOM.Object
$iphreeqc.LoadDatabase("phreeqc.dat") | Out-Null
$iphreeqc.OutputStringOn = $true

$in = ""
$in += "SOLUTION 1 Pure water     `n"
$in += "EQUILIBRIUM_PHASES 1      `n"
$in += "    Calcite 0 10          `n"

if ($iphreeqc.RunString($in) -eq 0) {
    Write-Host $iphreeqc.GetOutputString()
}

  JScript

var iphreeqc = new ActiveXObject("IPhreeqcCOM.Object");
iphreeqc.LoadDatabase("phreeqc.dat");
iphreeqc.OutputStringOn = true;

var input = new String("");
input += "SOLUTION 1 Pure water     \n";
input += "EQUILIBRIUM_PHASES 1      \n";
input += "    Calcite 0 10          \n";

if (iphreeqc.RunString(input) == 0) {
    WScript.Echo(iphreeqc.GetOutputString());
}

  VBScript (Deprecated)

Dim iphreeqc, sInput

Set iphreeqc = CreateObject("IPhreeqcCOM.Object")
iphreeqc.LoadDatabase("phreeqc.dat")
iphreeqc.OutputStringOn = True

sInput = ""
sInput = sInput & "SOLUTION 1 Pure water     " & vbNewLine
sInput = sInput & "EQUILIBRIUM_PHASES 1      " & vbNewLine
sInput = sInput & "    Calcite 0 10          " & vbNewLine

If (iphreeqc.RunString(sInput) = 0) Then
    WScript.Echo(iphreeqc.GetOutputString())
End If
